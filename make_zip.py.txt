# Write files
with open(os.path.join(base_dir, "index.html"), "w") as f: f.write(index_html)
with open(os.path.join(assets_dir, "style.css"), "w") as f: f.write(style_css)
with open(os.path.join(base_dir, "privacy.html"), "w") as f: f.write(privacy_html)
with open(os.path.join(base_dir, "readme.txt"), "w") as f: f.write(readme_txt)

# Create zip
zip_filename = "demurich-express-contact.zip"
with zipfile.ZipFile(zip_filename, "w", zipfile.ZIP_DEFLATED) as zipf:
    for root, _, files in os.walk(base_dir):
        for file in files:
            filepath = os.path.join(root, file)
            arcname = os.path.relpath(filepath, os.path.dirname(base_dir))
            zipf.write(filepath, arcname)

print(f"✅ Created {zip_filename}")
